        // Show gift card when button is clicked

        function showGiftCard() {

            document.getElementById('giftCard').style.display = 'block';

        // Save gift card functionality using canvas to generate an image

        function saveGiftCard() {

            var canvas = document.getElementById('giftCardCanvas');

            var ctx = canvas.getContext('2d');

            var giftCardDiv = document.getElementById('giftCard');


            // Set canvas size (same as gift card dimensions)

            canvas.width = giftCardDiv.offsetWidth;

            canvas.height = giftCardDiv.offsetHeight;


            // Draw gift card content on canvas

            ctx.fillStyle = 'rgba(255, 255, 255, 0.8)';

            ctx.fillRect(0, 0, canvas.width, canvas.height);


            // Add text for the gift card

            ctx.fillStyle = 'black';

            ctx.font = '20px Arial';

            ctx.fillText('礼物编号：NO.20250214520', 20, 30);

            ctx.fillText('礼物详情：Secret', 20, 70);

            ctx.fillText('请于恋爱存续期内凭此卡找dxx兑换', 20, 110);


            // Convert canvas content to image

            var imageUrl = canvas.toDataURL('image/png');


            // Create a temporary link to trigger download

            var link = document.createElement('a');

            link.href = imageUrl;

            link.download = 'Valentines_Gift_Card.png';

            link.click();

        }

    </script>
